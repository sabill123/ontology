# Letology UI/UX 설계 기획서
## Palantir AIP 수준의 데이터 인텔리전스 플랫폼

**Version**: 1.0
**Date**: 2026-01-26
**Author**: Claude AI Architect

---

## 1. Executive Summary

### 1.1 프로젝트 비전
Letology는 Palantir Foundry/AIP 수준의 **자율 데이터 온톨로지 발견 및 인텔리전스 플랫폼**입니다.
비전문가도 데이터를 업로드하면 자동으로 스키마, 관계, 인사이트를 발견하고 시각화합니다.

### 1.2 핵심 차별점
| Palantir AIP | Letology |
|--------------|----------|
| 수동 온톨로지 구축 | **자동 온톨로지 발견** |
| 전문가 필요 | **제로 설정** (Zero-Config) |
| 높은 진입 장벽 | **AI 가이드 UX** |
| 고가 라이선스 | **오픈소스 기반** |

### 6.3 그래프 시각화 구현

```typescript
// src/components/graph/DataMapGraph.tsx

import { useEffect, useRef } from 'react';
import cytoscape from 'cytoscape';
import dagre from 'cytoscape-dagre';

cytoscape.use(dagre);

interface DataMapGraphProps {
  nodes: VisNode[];
  edges: VisEdge[];
  onNodeSelect?: (nodeId: string) => void;
}

export function DataMapGraph({ nodes, edges, onNodeSelect }: DataMapGraphProps) {
  const containerRef = useRef<HTMLDivElement>(null);
  const cyRef = useRef<cytoscape.Core | null>(null);

  useEffect(() => {
    if (!containerRef.current) return;

    cyRef.current = cytoscape({
      container: containerRef.current,
      elements: {
        nodes: nodes.map(n => ({
          data: {
            id: n.id,
            label: n.label,
            type: n.type,
            ...n.metadata
          }
        })),
        edges: edges.map(e => ({
          data: {
            id: e.id,
            source: e.source,
            target: e.target,
            label: e.label,
            type: e.type
          }
        }))
      },
      style: [
        {
          selector: 'node[type="database"]',
          style: {
            'background-color': '#1a1a2e',
            'label': 'data(label)',
            'color': '#fff',
            'font-size': '14px',
            'width': 60,
            'height': 60,
            'shape': 'hexagon'
          }
        },
        {
          selector: 'node[type="table"]',
          style: {
            'background-color': '#0A84FF',
            'label': 'data(label)',
            'color': '#fff',
            'font-size': '12px',
            'width': 50,
            'height': 50
          }
        },
        {
          selector: 'edge[type="fk"]',
          style: {
            'line-color': '#64D2FF',
            'target-arrow-color': '#64D2FF',
            'target-arrow-shape': 'triangle',
            'curve-style': 'bezier',
            'label': 'data(label)',
            'font-size': '10px',
            'color': '#8E8E93'
          }
        }
      ],
      layout: {
        name: 'dagre',
        rankDir: 'TB',
        nodeSep: 50,
        rankSep: 100
      }
    });

    cyRef.current.on('tap', 'node', (evt) => {
      onNodeSelect?.(evt.target.id());
    });

    return () => {
      cyRef.current?.destroy();
    };
  }, [nodes, edges]);

  return (
    <div
      ref={containerRef}
      className="w-full h-full bg-[#1C1C1E] rounded-lg"
    />
  );
}
```

### 6.4 Python Sidecar 통합

```python
# python-sidecar/main.py

from fastapi import FastAPI, BackgroundTasks
from fastapi.middleware.cors import CORSMiddleware
import uvicorn
import sys
import os

# Add project root to path
sys.path.insert(0, os.path.dirname(os.path.dirname(__file__)))

from unified_pipeline import UnifiedPipeline
from unified_pipeline.visualization import PalantirStyleVisualizationGenerator

app = FastAPI(title="Letology Analysis Engine")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_methods=["*"],
    allow_headers=["*"],
)

# Global pipeline instance
pipeline = None

@app.on_event("startup")
async def startup():
    global pipeline
    pipeline = UnifiedPipeline()


@app.post("/analyze")
async def run_analysis(request: dict):
    """Run full autonomous analysis on uploaded data."""
    data_path = request.get("data_path")
    options = request.get("options", {})

    result = await pipeline.run_full_analysis(
        data_dir=data_path,
        enable_fk_detection=options.get("fk_detection", True),
        enable_ontology=options.get("ontology", True),
        enable_correlation=options.get("correlation", True),
        enable_insights=options.get("insights", True),
    )

    return result


@app.post("/visualize")
async def generate_visualization(request: dict):
    """Generate visualization data for frontend."""
    analysis_result = request.get("analysis_result")

    generator = PalantirStyleVisualizationGenerator()
    viz_data = generator.generate_all(
        tables_data=analysis_result.get("tables"),
        fk_results=analysis_result.get("fk_results"),
        ontology_data=analysis_result.get("ontology"),
        entity_data=analysis_result.get("entities"),
        insights=analysis_result.get("insights"),
    )

    return viz_data


@app.post("/simulate")
async def run_simulation(request: dict):
    """Run what-if simulation."""
    from unified_pipeline.autonomous.analysis.simulation_engine import SimulationEngine

    tables_data = request.get("tables_data")
    scenario = request.get("scenario")

    engine = SimulationEngine(tables_data)
    result = engine.run_what_if_scenario(scenario)

    return result


@app.post("/chat")
async def chat_with_ai(request: dict):
    """Natural language query interface."""
    query = request.get("query")
    context = request.get("context", {})

    # Use LLM to generate response
    response = await pipeline.natural_language_query(query, context)

    return response


if __name__ == "__main__":
    uvicorn.run(app, host="127.0.0.1", port=4001)
```
